% How to use these conditionals:
% - For a master CV, set \mastertrue.
% - For a ML Engineer/Data Scientist CV, set \masterfalse, and \mltrue
% - For a Software Developer CV, set \masterfalse, \softwaretrue
% - For a Web Developer CV, set \masterfalse, \webdevtrue

% Set the base conditionals
\webdevfalse
\mlfalse
\iosfalse
\desktopfalse

% Set the higher level conditionals
\softwarefalse

    \softwaretrue
\fi

\ifsoftware
    \desktoptrue
    \iostrue
    \webdevtrue
    \mlfalse
\fi

    \mltrue
\fi
